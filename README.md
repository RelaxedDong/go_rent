# 配置文件
```
appname = xxx
httpport = 8000
runmode = dev
copyrequestbody = true
logfilepath=logs/service.log
[dev]
httpaddr = 127.0.0.1
APPID=xxx # 小程序appid
APP_SECRET=xxx # 小程序 secret
[pro]
httpaddr = 127.0.0.1
APPID=xxx
APP_SECRET=xxx
```
